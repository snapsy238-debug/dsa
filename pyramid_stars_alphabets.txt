#include <stdio.h>

int main(){
    int n,i,j,space;
    printf("Enter number of rows: ");
    scanf("%d",&n);

    for(i=1;i<=n;i++){
        for(space=i;space<n;space++) printf(" ");
        for(j=1;j<=2*i-1;j++) printf("*");
        printf("\n");
    }

    char ch='A';
    for(i=1;i<=n;i++){
        for(space=i;space<n;space++) printf(" ");
        for(j=1;j<=2*i-1;j++) printf("%c", ch);
        printf("\n");
        ch++;
    }
    return 0;
}
