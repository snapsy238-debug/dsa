#include <stdio.h>
#define SIZE 5
int stack[SIZE], top=-1;

void push(int val){
    if(top<SIZE-1) stack[++top]=val;
}

void pop(){
    if(top!=-1) top--;
}

void display(){
    for(int i=top;i>=0;i--) printf("%d ",stack[i]);
}

int main(){
    int ch,val;
    do{
        scanf("%d",&ch);
        switch(ch){
            case 1: scanf("%d",&val);push(val);break;
            case 2: pop();break;
            case 3: display();break;
        }
    }while(ch!=4);
    return 0;
}
