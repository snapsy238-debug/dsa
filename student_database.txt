#include <stdio.h>
#include <string.h>

struct student{
    int roll;
    char name[20],dept[20],subject[20];
    int marks;
} s[10];
int n=0;

void add(){
    scanf("%d%s%s%s%d",&s[n].roll,s[n].name,s[n].dept,s[n].subject,&s[n].marks);
    n++;
}

void display(){
    for(int i=0;i<n;i++)
        printf("%d %s %s %s %d\n",s[i].roll,s[i].name,s[i].dept,s[i].subject,s[i].marks);
}

void search(){
    int r;scanf("%d",&r);
    for(int i=0;i<n;i++)
        if(s[i].roll==r){
            printf("%s %s %s %d\n",s[i].name,s[i].dept,s[i].subject,s[i].marks);
            return;
        }
}

void delete(){
    int r;scanf("%d",&r);
    for(int i=0;i<n;i++)
        if(s[i].roll==r){
            for(int j=i;j<n-1;j++) s[j]=s[j+1];
            n--;return;
        }
}

int main(){
    int ch;
    do{
        scanf("%d",&ch);
        switch(ch){
            case 1:add();break;
            case 2:delete();break;
            case 3:search();break;
            case 4:display();break;
        }
    }while(ch!=5);
    return 0;
}
