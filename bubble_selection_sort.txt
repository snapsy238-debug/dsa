#include <stdio.h>

void bubble(int a[],int n){
    for(int i=0;i<n-1;i++)
        for(int j=0;j<n-i-1;j++)
            if(a[j]>a[j+1]){
                int t=a[j];a[j]=a[j+1];a[j+1]=t;
            }
}

void selection(int a[],int n){
    for(int i=0;i<n-1;i++){
        int min=i;
        for(int j=i+1;j<n;j++)
            if(a[j]<a[min]) min=j;
        int t=a[i];a[i]=a[min];a[min]=t;
    }
}

int main(){
    int a[5]={5,3,2,4,1};
    bubble(a,5);
    for(int i=0;i<5;i++) printf("%d ",a[i]);

    int b[5]={5,3,2,4,1};
    selection(b,5);
    for(int i=0;i<5;i++) printf("%d ",b[i]);
    return 0;
}
